docker build -t django-docker-image . #создание образа
docker run -d --name django-docker-container -p 8001:8000 django-docker-image #создание контейнера из образа
docker run -d --name django-docker-container -v .:/app -p 8001:8000 django-docker-image #создание контейнера из образа с последующей синхронизацией
docker stop django-docker-container #остановить контейнер
docker rm django-docker-container #удалить контейнер
docker compose up -d #запуск compose файла
docker compose down #остановить compose файл
